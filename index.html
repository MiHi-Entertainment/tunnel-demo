  <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r148/three.min.js"></script>

  <script>

    // Basic Three.js Scene

    let scene = new THREE.Scene();

    let camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);

    let renderer = new THREE.WebGLRenderer({ antialias: true });

    renderer.setSize(window.innerWidth, window.innerHeight);

    document.body.appendChild(renderer.domElement);


    // Tunnel Geometry

    let geometry = new THREE.CylinderGeometry(5, 5, 100, 32, 32, true);

    let material = new THREE.MeshBasicMaterial({ color: 0x00ffff, wireframe: true, side: THREE.BackSide });

    let tunnel = new THREE.Mesh(geometry, material);

    scene.add(tunnel);


    camera.position.z = 0;


    // Scroll-based Z movement

    let scrollY = 0;

    window.addEventListener("scroll", () => {

      scrollY = window.scrollY;

    });


    function animate() {

      requestAnimationFrame(animate);

      tunnel.rotation.z += 0.002;

      camera.position.z = scrollY * 0.01; // Scroll movement

      renderer.render(scene, camera);

    }


    animate();

  </script>
